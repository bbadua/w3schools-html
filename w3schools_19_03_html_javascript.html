html javascript

changing image src and alt



<script>

function picChanger(val){
	var src;
	var alt;

	if(val == 200){
		src = "200px.jpg";
		alt = "200 px image";
	} else {
		src = "400px.jpg";
		alt = "400 px image";
	}
	
	document.getElementById("img01").src = src;
	document.getElementById("img01").alt = alt;
}

</script>

...

<img id="img01" src="200px.jpg" alt="200 px image"><br>

<button onclick="picChanger(200);">200</button>
<button onclick="picChanger(400);">400</button>
