<script>
function hideCityElements(){
	var x = document.getElementsByClassName("city");		//get collection of elements with class 'city'

	for(var i=0;i<x.length;i++){					//iterate through collection
		x[i].style.display = "none";				//for each item, set display to "none"
	}	
}

</script>
